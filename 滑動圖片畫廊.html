swiperjs 官網

程式碼範例

                const swiper = new Swiper('.swiper', {
                    slidesPerView: 1, // 每次顯示的幻燈片數
                    spaceBetween: 10, // 幻燈片之間的間距
                    navigation: {
                        nextEl: '.swiper-button-next',
                        prevEl: '.swiper-button-prev',
                    },
                    scrollbar: {
                        el: '.swiper-scrollbar',
                        hide: true,
                    },
                });
