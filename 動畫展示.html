Lottie.js 官網

程式碼範例

                <!-- Lottie.js -->
                <script src="https://cdnjs.cloudflare.com/ajax/libs/lottie-web/5.7.13/lottie.min.js"></script>
            
                <!-- Lottie 動畫初始化 -->
                <script>
                    lottie.loadAnimation({
                        container: document.getElementById('lottie-animation'),
                        renderer: 'svg',
                        loop: true,
                        autoplay: true,
                        path: 'YOUR_JSON_FILE_URL.json'
                    });
                </script>
